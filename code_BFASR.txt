
// Oak Leaf-Roller Moth Study - Sentinel-2 Pipeline

// 1. Define study area
var studyArea = ee.FeatureCollection("projects/ee-hbig6399/assets/StudyArea");

// 2. Center map
Map.centerObject(studyArea, 9);
Map.addLayer(studyArea, {}, 'Study Area');

// 3. Define years and date range
var startYear = 2018;
var endYear = 2024;
var startMonth = 3, startDay = 21;
var endMonth = 10, endDay = 22;

// 4. Cloud masking function using S2cloudless
function maskS2clouds(image) {
  var cloudProb = ee.ImageCollection('COPERNICUS/S2_CLOUD_PROBABILITY')
                    .filterBounds(studyArea)
                    .filterDate(image.date(), image.date().advance(1,'day'))
                    .first();
  var cloudMask = cloudProb.select('probability').lt(30);
  
  // Apply 200m buffer to remove cloud edges/shadows
  var bufferedMask = cloudMask.focal_min(20).focal_max(20);
  
  return image.updateMask(bufferedMask);
}

// 5. Vegetation indices calculation
function addIndices(img) {
  var ndvi = img.normalizedDifference(['B8', 'B4']).rename('NDVI');
  var evi = img.expression(
    '2.5 * ((NIR - RED) / (NIR + 6*RED - 7.5*BLUE + 1))',
    {
      NIR: img.select('B8'),
      RED: img.select('B4'),
      BLUE: img.select('B2')
    }
  ).rename('EVI');
  var ndwi = img.normalizedDifference(['B3', 'B8']).rename('NDWI');
  return img.addBands([ndvi, evi, ndwi]);
}

// 6. Loop over years
for (var year = startYear; year <= endYear; year++) {
  
  var startDate = ee.Date.fromYMD(year, startMonth, startDay);
  var endDate = ee.Date.fromYMD(year, endMonth, endDay);
  
  // 7. Load Sentinel-2 L2A collection
  var s2 = ee.ImageCollection('COPERNICUS/S2_SR_HARMONIZED')
              .filterBounds(studyArea)
              .filterDate(startDate, endDate)
              .filter(ee.Filter.lt('CLOUDY_PIXEL_PERCENTAGE', 100)) // all images, masking later
              .map(maskS2clouds)
              .map(addIndices);
  
  // 8. Median composite for visualization
  var medianImage = s2.median();
  
  // 9. Threshold-based decline map (for visualization only)
  var ndviThreshold = 0.7;
  var eviThreshold = 0.7;
  var ndwiThreshold = 0.3;
  
  var decline = medianImage.select('NDVI').lt(ndviThreshold)
                .or(medianImage.select('EVI').lt(eviThreshold))
                .or(medianImage.select('NDWI').lt(ndwiThreshold))
                .selfMask();
  
  Map.addLayer(decline, {palette: ['red']}, 'Decline Areas ' + year);
  
  // 10. Export decline GeoTIFF
  Export.image.toDrive({
    image: decline,
    description: 'Decline_Map_' + year,
    folder: 'GEE_Exports',
    fileNamePrefix: 'Decline_Map_' + year,
    region: studyArea.geometry(),
    scale: 10,
    maxPixels: 1e13
  });
  
  // 11. Daily mean indices (ready for BFAST in R/Python)
  var dailyIndices = s2.map(function(img) {
    var date = img.date().format('YYYY-MM-dd');
    var meanDict = img.reduceRegion({
      reducer: ee.Reducer.mean(),
      geometry: studyArea,
      scale: 10,
      maxPixels: 1e9
    });
    return ee.Feature(null, {
      'date': date,
      'NDVI': meanDict.get('NDVI'),
      'EVI': meanDict.get('EVI'),
      'NDWI': meanDict.get('NDWI')
    });
  });
  
  // 12. Export daily indices CSV
  Export.table.toDrive({
    collection: dailyIndices,
    description: 'Daily_Indices_' + year,
    fileFormat: 'CSV',
    folder: 'GEE_Exports',
    fileNamePrefix: 'Daily_Indices_' + year
  });
}
